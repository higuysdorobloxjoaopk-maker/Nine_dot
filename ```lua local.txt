local activeTouches = {}
local simulatedKeys = {}

local function resetSimulatedKeys()
    for kc, _ in pairs(simulatedKeys) do
        VirtualInputManager:SendKeyEvent(false, kc, false, game)
        if buttonMapping[kc] then
            buttonMapping[kc].BackgroundTransparency = 1
        end
    end
    table.clear(simulatedKeys)
    table.clear(activeTouches)
end

local function updateButtonVisuals(keyCode, isPressed)
    if editActive then return end
    local button = buttonMapping[keyCode]
    if button then
        button.BackgroundTransparency = isPressed and 0.5 or 1
    end
end

local function simulateKeyPress(keyCode, isPressed)
    if editActive then return end
    VirtualInputManager:SendKeyEvent(isPressed, keyCode, false, game)
end

local function checkTouches()
    if editActive then return end
    local touchesOnButtons = {}

    for touch in pairs(activeTouches) do  
        local touchPosition = touch.Position  
        for keyCode, button in pairs(buttonMapping) do  
            local absolutePosition = button.AbsolutePosition  
            local absoluteSize = button.AbsoluteSize  
              
            local isInsideX = touchPosition.X >= absolutePosition.X and touchPosition.X <= absolutePosition.X + absoluteSize.X  
            local isInsideY = touchPosition.Y >= absolutePosition.Y and touchPosition.Y <= absolutePosition.Y + absoluteSize.Y  
              
            if isInsideX and isInsideY then  
                touchesOnButtons[keyCode] = true  
            end  
        end  
    end  
      
    for keyCode, button in pairs(buttonMapping) do  
        local isBeingTouched = touchesOnButtons[keyCode]  
        local wasSimulated = simulatedKeys[keyCode]  
          
        if isBeingTouched and not wasSimulated then  
            simulateKeyPress(keyCode, true)  
            updateButtonVisuals(keyCode, true)  
            simulatedKeys[keyCode] = true  
        elseif not isBeingTouched and wasSimulated then  
            simulateKeyPress(keyCode, false)  
            updateButtonVisuals(keyCode, false)  
            simulatedKeys[keyCode] = nil  
        end  
    end
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if editActive then return end
    if input.UserInputType == Enum.UserInputType.Touch then
        activeTouches[input] = true
        checkTouches()
    elseif input.UserInputType == Enum.UserInputType.Keyboard and not gameProcessed then
        updateButtonVisuals(input.KeyCode, true)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if editActive then return end
    if input.UserInputType == Enum.UserInputType.Touch then
        checkTouches()
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if editActive then return end
    if input.UserInputType == Enum.UserInputType.Touch then
        activeTouches[input] = nil
        checkTouches()
    elseif input.UserInputType == Enum.UserInputType.Keyboard then
        updateButtonVisuals(input.KeyCode, false)
    end
end)

for keyCode, button in pairs(buttonMapping) do
    button.MouseButton1Down:Connect(function()
        if editActive then return end
        simulateKeyPress(keyCode, true)
        updateButtonVisuals(keyCode, true)
        simulatedKeys[keyCode] = true
    end)

    button.MouseButton1Up:Connect(function()  
        if editActive then return end
        simulateKeyPress(keyCode, false)  
        updateButtonVisuals(keyCode, false)  
        simulatedKeys[keyCode] = nil  
    end)  
      
    button.MouseLeave:Connect(function()  
        if editActive then return end
        if simulatedKeys[keyCode] then  
            simulateKeyPress(keyCode, false)  
            updateButtonVisuals(keyCode, false)  
            simulatedKeys[keyCode] = nil  
        end  
    end)  
      
    button.MouseEnter:Connect(function()  
        if editActive then return end
        if UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then  
            simulateKeyPress(keyCode, true)  
            updateButtonVisuals(keyCode, true)  
            simulatedKeys[keyCode] = true  
        end  
    end)
end

local function hexToColor3(hexStr)
    hexStr = hexStr:gsub("#", "")
    local success, c3 = pcall(function() return Color3.fromHex(hexStr) end)
    return success and c3 or Color3.fromRGB(0,0,0)
end

local function applyModSettings(settings)
    local bgC3 = settings.BgMode == "Black" and Color3.fromRGB(50, 50, 50) or Color3.fromRGB(255, 255, 255)
    local txtC3 = hexToColor3(settings.TextHex)

    -- Aplicar Background com base no ImageLabel das teclas
    for _, obj in ipairs(screenGui:GetDescendants()) do
        if obj:IsA("ImageLabel") and obj.Image == "rbxassetid://123775806555660" and not obj:IsDescendantOf(UI_CreateMenu) and not obj:IsDescendantOf(UI_ModMenu) then
            -- Só não afeta as teclas de clique ativo/feedback
            if not obj:GetAttribute("ActiveAutoClickVisual") then
                obj.ImageColor3 = bgC3
            end
        end
    end

    if activeRainbowConnection then
        activeRainbowConnection:Disconnect()
        activeRainbowConnection = nil
    end

    if settings.Rainbow then
        activeRainbowConnection = RunService.RenderStepped:Connect(function()
            local hue = tick() % 3 / 3
            local rainbowColor = Color3.fromHSV(hue, 1, 1)
            for kc, btn in pairs(buttonMapping) do
                if btn then btn.TextColor3 = rainbowColor end
            end
        end)
    else
        for kc, btn in pairs(buttonMapping) do
            if btn then btn.TextColor3 = txtC3 end
        end
    end
end

local function cleanButtonMappings(targetFrame)
    for kc, btn in pairs(buttonMapping) do
        if btn:IsDescendantOf(targetFrame) then
            buttonMapping[kc] = nil
        end
    end
end

-- Parser de texto para KeyCode, aceita Esc, ., ,, *, etc.
local function parseKeyCode(str)
    if type(str) ~= "string" or str == "" then return nil end
    
    local map = {
        ["."] = "Period", [","] = "Comma", ["/"] = "Slash", ["\\"] = "BackSlash",
        ["'"] = "Quote", [";"] = "Semicolon", ["="] = "Equals", ["-"] = "Minus",
        ["["] = "LeftBracket", ["]"] = "RightBracket", ["`"] = "Backquote",
        ["*"] = "Asterisk", ["+"] = "Plus", [" "] = "Space", ["_"] = "Underscore",
        ["esc"] = "Escape", ["escape"] = "Escape", ["alt"] = "LeftAlt",
        ["ctrl"] = "LeftControl", ["shift"] = "LeftShift", ["tab"] = "Tab",
        ["enter"] = "Return", ["return"] = "Return", ["backspace"] = "Backspace",
        ["del"] = "Delete", ["delete"] = "Delete", ["up"] = "Up", ["down"] = "Down",
        ["left"] = "Left", ["right"] = "Right", ["f1"] = "F1", ["f2"] = "F2",
        ["f3"] = "F3", ["f4"] = "F4", ["f5"] = "F5", ["f6"] = "F6", ["f7"] = "F7",
        ["f8"] = "F8", ["f9"] = "F9", ["f10"] = "F10", ["f11"] = "F11", ["f12"] = "F12"
    }
    
    local lowerStr = string.lower(str)
    local mapped = map[lowerStr] or str
    
    if #mapped == 1 then
        mapped = string.upper(mapped)
    end
    
    local success, val = pcall(function() return Enum.KeyCode[mapped] end)
    if success then return val end
    return nil
end

local function BindCustomButtonEvents(keyCode, button)
    button.MouseButton1Down:Connect(function()
        if editActive then return end
        simulateKeyPress(keyCode, true)
        updateButtonVisuals(keyCode, true)
        simulatedKeys[keyCode] = true
    end)
    button.MouseButton1Up:Connect(function()  
        if editActive then return end
        simulateKeyPress(keyCode, false)  
        updateButtonVisuals(keyCode, false)  
        simulatedKeys[keyCode] = nil  
    end)  
    button.MouseLeave:Connect(function()  
        if editActive then return end
        if simulatedKeys[keyCode] then  
            simulateKeyPress(keyCode, false)  
            updateButtonVisuals(keyCode, false)  
            simulatedKeys[keyCode] = nil  
        end  
    end)  
    button.MouseEnter:Connect(function()  
        if editActive then return end
        if UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then  
            simulateKeyPress(keyCode, true)  
            updateButtonVisuals(keyCode, true)  
            simulatedKeys[keyCode] = true  
        end  
    end)
end

-- Nova função unificada para criar Elementos Customizados (Teclas, Atalhos, AutoClick)
local function CreateCustomElement(data)
    local keyName = data.Name
    local eType = data.Type or "Key"
    local position = UDim2.new(unpack(data.Position))

    local kContainer = Instance.new("Frame")
    kContainer.Name = "Custom_" .. keyName .. "_" .. tostring(tick())
    kContainer.BackgroundTransparency = 1
    kContainer.Position = position
    kContainer.Parent = screenGui

    -- Largura automática dependendo do tamanho do texto
    local textBounds = TextService:GetTextSize(keyName, 17, Enum.Font.SourceSansBold, Vector2.new(1000, 45))
    local btnWidth = math.max(45, textBounds.X + 24)
    kContainer.Size = UDim2.new(0, btnWidth, 0, 45)

    local kImage = Instance.new("ImageLabel")
    kImage.Size = UDim2.new(1, 0, 1, 0)
    kImage.BackgroundTransparency = 1
    kImage.Image = "rbxassetid://123775806555660"
    kImage.Parent = kContainer

    local kButton = Instance.new("TextButton")
    kButton.Name = keyName
    kButton.Text = keyName
    kButton.Size = UDim2.new(1, 0, 1, 0)
    kButton.BackgroundTransparency = 1
    kButton.FontFace = Font.new("rbxasset://fonts/families/SourceSansPro.json", Enum.FontWeight.Bold, Enum.FontStyle.Normal)
    kButton.TextSize = 17
    kButton.TextColor3 = Color3.fromRGB(0, 0, 0)
    kButton.Parent = kImage

    local customElementData = {
        frame = kContainer,
        id = kContainer.Name,
        isCustom = true,
        data = data,
        button = kButton -- Armazena a referência para desativá-la no modo de edição
    }

    local targetBall = nil
    if eType == "Tap" then
        targetBall = Instance.new("Frame")
        targetBall.Name = "Target_" .. kContainer.Name
        targetBall.Size = UDim2.new(0, 30, 0, 30)
        if data.TargetPos then
            targetBall.Position = UDim2.new(unpack(data.TargetPos))
        else
            -- Coloca a bola amarela acima do botão inicialmente usando o Scale e Offset juntos
            targetBall.Position = UDim2.new(position.X.Scale, position.X.Offset, position.Y.Scale, position.Y.Offset - 60)
        end
        targetBall.BackgroundColor3 = Color3.fromRGB(255, 255, 0)
        targetBall.BackgroundTransparency = editActive and 0.5 or 1
        targetBall.Visible = editActive -- Invisivel completo fora da edição
        targetBall.Parent = screenGui
        
        local corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(1, 0)
        corner.Parent = targetBall
        
        local dragBall = Instance.new("UIDragDetector")
        dragBall.Enabled = editActive
        dragBall.BoundingUI = dragBoundary
        dragBall.Parent = targetBall
        
        customElementData.targetBall = targetBall
        customElementData.dragBall = dragBall
        
        -- Eventos para Taps / AutoClick
        local isClicking = false
        local clickCoro = nil
        
        kButton.MouseButton1Down:Connect(function()
            if editActive then return end
            kImage.ImageTransparency = 0.5
            
            local tAbs = targetBall.AbsolutePosition
            local tSize = targetBall.AbsoluteSize
            local cx = tAbs.X + tSize.X / 2
            local cy = tAbs.Y + tSize.Y / 2
            
            if keyName == "AutoClick" then
                isClicking = not isClicking
                if isClicking then
                    kImage:SetAttribute("ActiveAutoClickVisual", true)
                    kImage.ImageColor3 = Color3.fromRGB(100, 255, 100) -- visual verde
                    clickCoro = task.spawn(function()
                        local cps = data.CPS or 10
                        local delayTime = 1 / cps
                        while isClicking do
                            VirtualInputManager:SendMouseButtonEvent(cx, cy, 0, true, game, 1)
                            task.wait(0.02)
                            VirtualInputManager:SendMouseButtonEvent(cx, cy, 0, false, game, 1)
                            task.wait(math.max(0, delayTime - 0.02))
                        end
                    end)
                else
                    kImage:SetAttribute("ActiveAutoClickVisual", nil)
                    applyModSettings(currentModSettings) -- Restaura visual padrão
                    if clickCoro then task.cancel(clickCoro) end
                end
            else
                VirtualInputManager:SendMouseButtonEvent(cx, cy, 0, true, game, 1)
            end
        end)
        
        kButton.MouseButton1Up:Connect(function()
            if editActive then return end
            kImage.ImageTransparency = 0
            if keyName ~= "AutoClick" then
                local tAbs = targetBall.AbsolutePosition
                local tSize = targetBall.AbsoluteSize
                local cx = tAbs.X + tSize.X / 2
                local cy = tAbs.Y + tSize.Y / 2
                VirtualInputManager:SendMouseButtonEvent(cx, cy, 0, false, game, 1)
            end
        end)
        
        kButton.MouseLeave:Connect(function()
            if editActive then return end
            kImage.ImageTransparency = 0
            if keyName ~= "AutoClick" then
                local tAbs = targetBall.AbsolutePosition
                local tSize = targetBall.AbsoluteSize
                local cx = tAbs.X + tSize.X / 2
                local cy = tAbs.Y + tSize.Y / 2
                VirtualInputManager:SendMouseButtonEvent(cx, cy, 0, false, game, 1)
            end
        end)
        
    else
        -- Configuração normal para teclado
        local kc = parseKeyCode(data.KeyCodeStr)
        if kc then
            buttonMapping[kc] = kButton
            BindCustomButtonEvents(kc, kButton)
        end
    end

    -- Lógica de arrasto e edição para o botão
    local dragBtn = Instance.new("UIDragDetector")
    dragBtn.Enabled = editActive
    dragBtn.BoundingUI = dragBoundary
    dragBtn.Parent = kContainer
    customElementData.drag = dragBtn
    
    local lastClick = 0
    local startPos = Vector3.new()
    
    kContainer.InputBegan:Connect(function(input)
        if editActive and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
            startPos = input.Position
            local now = tick()
            if now - lastClick < 0.3 then -- Duplo clique
                -- Deletar Componente
                if eType == "Tap" and targetBall then targetBall:Destroy() end
                if eType == "Key" then
                    local kc = parseKeyCode(data.KeyCodeStr)
                    if kc then buttonMapping[kc] = nil end
                end
                kContainer:Destroy()
                for i, v in ipairs(draggables) do
                    if v.frame == kContainer then
                        table.remove(draggables, i)
                        break
                    end
                end
            end
            lastClick = now
        end
    end)
    
    kContainer.InputEnded:Connect(function(input)
        if editActive and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1) then
            if kContainer.Parent then
                local dist = (input.Position - startPos).Magnitude
                -- Se não arrastou (apenas clicou soltando o dedo)
                if dist < 10 and keyName == "AutoClick" and eType == "Tap" then
                    OpenCPSMenu(customElementData)
                end
            end
        end
    end)

    table.insert(draggables, customElementData)
    applyModSettings(currentModSettings)
    
    if editActive then
        kContainer.BackgroundTransparency = 0.5
        kContainer.BackgroundColor3 = Color3.fromRGB(100, 200, 255)
        kButton.Interactable = false
        kButton.Active = false
    end
    
    return customElementData
end

local configPath = "TouchGui/Config.json"
local PlaceIdStr = tostring(game.PlaceId)

local function ReadConfigFile()
    if isfile and isfile(configPath) then
        local s, r = pcall(readfile, configPath)
        if s then
            local ds, decoded = pcall(function() return HttpService:JSONDecode(r) end)
            if ds and type(decoded) == "table" then return decoded end
        end
    end
    return {}
end

local function SaveData()
    if isfolder and not isfolder("TouchGui") then pcall(makefolder, "TouchGui") end
    local allData = ReadConfigFile()
    local layoutData = { CustomElements = {}, ModSettings = currentModSettings }

    for _, item in ipairs(draggables) do
        if item.frame and item.frame.Parent then
            local pos = item.frame.Position
            local posTable = {pos.X.Scale, pos.X.Offset, pos.Y.Scale, pos.Y.Offset}
            
            if item.isCustom then
                local tPosTable = nil
                if item.targetBall and item.targetBall.Parent then
                    local tPos = item.targetBall.Position
                    tPosTable = {tPos.X.Scale, tPos.X.Offset, tPos.Y.Scale, tPos.Y.Offset}
                    item.data.TargetPos = tPosTable
                end
                item.data.Position = posTable
                table.insert(layoutData.CustomElements, item.data)
            else
                layoutData[item.id] = posTable
            end
        end
    end

    allData[PlaceIdStr] = layoutData
    if writefile then
        local s, r = pcall(function() return HttpService:JSONEncode(allData) end)
        if s then pcall(writefile, configPath, r) end
    end
end

local function LoadAndApply()
    for i = #draggables, 1, -1 do
        local item = draggables[i]
        if item.isCustom then
            if item.frame then item.frame:Destroy() end
            if item.targetBall then item.targetBall:Destroy() end
            if item.data and item.data.Type == "Key" then
                local kc = parseKeyCode(item.data.KeyCodeStr)
                if kc then buttonMapping[kc] = nil end
            end
            table.remove(draggables, i)
        end
    end

    local allData = ReadConfigFile()
    local myData = allData[PlaceIdStr]

    if myData then
        if myData.WASD then wasdContainer.Position = UDim2.new(unpack(myData.WASD)) end
        if myData.Space then spaceContainer.Position = UDim2.new(unpack(myData.Space)) end
        if myData.ExtraKeys then 
            if extraKeysContainer.Parent then
                extraKeysContainer.Position = UDim2.new(unpack(myData.ExtraKeys)) 
            end
        else
            if extraKeysContainer.Parent then
                cleanButtonMappings(extraKeysContainer)
                extraKeysContainer:Destroy()
            end
        end
        
        if myData.CustomElements then
            for _, kData in ipairs(myData.CustomElements) do
                CreateCustomElement(kData)
            end
        elseif myData.CustomKeys then -- Compatibilidade com dados antigos
            for _, k in ipairs(myData.CustomKeys) do
                CreateCustomElement({
                    Name = k.Name,
                    Type = "Key",
                    KeyCodeStr = k.KeyCode,
                    Position = k.Position,
                    CPS = 10
                })
            end
        end
        
        if myData.ModSettings then
            currentModSettings = myData.ModSettings
        end
    else
        wasdContainer.Position = UDim2.new(0, 6, 0, 196)
        if extraKeysContainer.Parent then extraKeysContainer.Position = UDim2.new(0, 628, 0, 195) end
        spaceContainer.Position = UDim2.new(0, 588, 0, 222)
    end
    
    applyModSettings(currentModSettings)
end

local function setDraggablesState(active)
    for _, item in ipairs(draggables) do
        if item.frame and item.frame.Parent then
            if item.drag then item.drag.Enabled = active end
            if active then
                item.frame.BackgroundTransparency = 0.5
                item.frame.BackgroundColor3 = Color3.fromRGB(100, 200, 255)
            else
                item.frame.BackgroundTransparency = 1
            end
            
            -- Desativa o TextButton temporariamente no modo Edit para deixar o UIDragDetector funcionar no Tap/AutoClick
            if item.button then
                item.button.Interactable = not active
                item.button.Active = not active
            end
            
            if item.targetBall then
                item.targetBall.BackgroundTransparency = active and 0.5 or 1
                item.targetBall.Visible = active -- Alterna a visibilidade completa da bolinha amarela
                if item.dragBall then item.dragBall.Enabled = active end
            end
        end
    end
    
    for kc, btn in pairs(buttonMapping) do
        if btn and btn.Parent then
            btn.Interactable = not active
            btn.Active = not active
        end
    end
end
